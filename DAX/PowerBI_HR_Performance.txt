--Table

Employee_Calculated =
SELECTCOLUMNS(
    Employees,
    "EmployeeID", Employees[EmployeeID],
    "FullName", TRIM(COALESCE(Employees[FirstName], "") & " " & COALESCE(Employees[LastName], "")),
    "Age", DATEDIFF(Employees[BirthDate], TODAY(), YEAR),
    "YearsInCompany", DATEDIFF(Employees[HireDate], TODAY(), YEAR)
)
